
conquer_scotland = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	target_provinces = {
		OR = {
			region = highlands
			region = lowlands
		}
		owned_by = SCO
	}
	allow = {
		OR = {
			tag = ENG
			tag = GBR
		}
		is_subject = no
		is_year = 1603
		SCO = {
			is_subject = no
			OR = {
				highlands = { owned_by = SCO }
				lowlands = { owned_by = SCO }
			}
		}
		NOT = { has_opinion = { who = SCO value = 50 } }
	}
	abort = {
		OR = {
			is_subject = yes
			SCO = { is_subject_of = ROOT }
			any_target_province = {
				NOT = { owned_by = ROOT }
				NOT = { owned_by = SCO }
			}
		}
	}
	success = {
		NOT = { exists = SCO }
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 2
			mil = 4
		}
		modifier = {
			factor = 2
			mil = 5
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			remove_claim = ROOT
		}
	}
	effect = {
		add_prestige = 10
	}
}

			}

}
