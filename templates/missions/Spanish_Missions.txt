finish_reconquista = {

	type = country

	category = MIL
	ai_mission = yes
	
	target_provinces = {
		region = iberian_peninsula
		owner = { religion_group = muslim }
	}
	allow = {
		NOT = { is_year = 1600 }
		OR = {
			tag = CAS
			tag = SPA
		}
		is_subject = no
		is_at_war = no
		religion = catholic
		iberian_peninsula = {
			owned_by = ROOT
		}
		iberian_peninsula = {
			owner = { religion_group = muslim }
		}
	}
	abort = {
		OR = {
			NOT = { religion = catholic }
			NOT = { iberian_peninsula = { owned_by = ROOT } }
		}
	}
	success = {
		iberian_peninsula = {
			type = all
			owner = { NOT = { religion_group = muslim } }
		}
		all_target_province = {
			owned_by = ROOT
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			remove_claim = ROOT
		}
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 10
			exists = GRA
			iberian_peninsula = {
				owned_by = GRA
			}
		}
	}
	effect = {
		add_army_tradition = 15
		add_war_exhaustion = -4
	}
}
