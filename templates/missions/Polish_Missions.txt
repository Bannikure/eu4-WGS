
poland_mazovia_relations = {
	
	type = country

	category = DIP
	
	allow = {
		tag = POL
		exists = MAZ
		is_subject = no
		government = monarchy
		NOT = { war_with = MAZ }
		NOT = { marriage_with = MAZ }
		MAZ = {
			is_subject = no
			government = monarchy
		}
	}
	abort = {
		OR = {
			is_subject = yes
			NOT = { government = monarchy }
			NOT = { exists = MAZ }
			war_with = MAZ
			MAZ = {
				OR = {
					is_subject = yes
					NOT = { government = monarchy }
				}
			}
		}
	}
	success = {
		marriage_with = MAZ
		MAZ = { has_opinion = { who = POL value = 100 } }
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = MAZ value = 0 } }
		}
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = MAZ value = -50 } }
		}
	}
	effect = {
		add_prestige = 5
	}
}


vassalize_mazovia = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	allow = {
		tag = POL
		exists = MAZ
		is_subject = no
		MAZ = {
			is_subject = no
			is_neighbor_of = ROOT
			NOT = { num_of_cities = ROOT }
		}
	}
	abort = {
		OR = {
			is_subject = yes
			NOT = { exists = MAZ }
			MAZ = { is_subject = yes }
		}
	}
	success = {
		MAZ = { vassal_of = POL }
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 2
			marriage_with = MAZ
		}
		modifier = {
			factor = 2
			has_opinion = { who = MAZ value = 100 }
		}
	}
	immediate = {
		add_casus_belli = {
			type = cb_vassalize_mission
			months = 120
			target = MAZ
		}
	}
	abort_effect = {
		remove_casus_belli = {
			type = cb_vassalize_mission
			target = MAZ
		}
	}
	effect = {
		add_prestige = 10
		hidden_effect = {
			remove_casus_belli = {
				type = cb_vassalize_mission
				target = MAZ
			}
		}
	}
}

		}
	}

		is_year = 1700
		exists = HAB
		is_neighbor_of = HAB
		any_owned_province = {
			NOT = { has_building = fort3 }
			can_build = fort3

		}
	}
	abort = {
		HAB = { NOT = { is_neighbor_of = ROOT } }

			factor = 2

		}
	}
	effect = {

	}
}
