lithuania_poland_relations = {
	
	type = country

	category = DIP
	
	allow = {
		tag = LIT
		exists = POL
		is_subject = no
		government = monarchy
		NOT = { war_with = POL }
		NOT = { marriage_with = POL }
		POL = {
			is_subject = no
			government = monarchy
		}
	}
	abort = {
		OR = {
			is_subject = yes
			NOT = { government = monarchy }
			war_with = POL
			NOT = { exists = POL }
			POL = {
				OR = {
					is_subject = yes
					NOT = { government = monarchy }
				}
			}
		}
	}
	success = {
		marriage_with = POL
		POL = { has_opinion = { who = LIT value = 100 } }
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = POL value = 0 } }
		}
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = POL value = -100 } }
		}
	}
	effect = {
		add_stability = 1
	}
}
