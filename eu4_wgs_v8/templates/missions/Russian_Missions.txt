	type = country
	
	category = MIL
	ai_mission = yes

		region = finnish_region
		owned_by = SWE
	}
	allow = {
		tag = RUS
		is_subject = no
		is_neighbor_of = SWE
		NOT = { war_with = SWE }
		finnish_region = { owned_by = SWE }
		NOT = { finnish_region = { owned_by = RUS } }
	}
	abort = {
		OR = {
			is_subject = yes
			any_target_province = {
				NOT = { owned_by = ROOT }
				NOT = { owned_by = SWE }
			}
		}
	}
	success = {
		NOT = { war_with = SWE }
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 2
			mil = 4
		}	
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			remove_claim = ROOT
		}
	}
	effect = {
		add_adm_power = 100
	}
}
