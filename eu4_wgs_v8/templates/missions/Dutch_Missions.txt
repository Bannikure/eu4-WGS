defend_the_lowlands = {
	
	type = country

	category = MIL
	
	target_provinces = {
		region = the_low_countries
		owned_by = SPA
	}
	allow = {
		tag = NED
		NOT = { is_year = 1649 }
		is_neighbor_of = SPA
		the_low_countries = { owned_by = SPA }
	}
	abort = {
		is_subject = yes
	}
	success = {
		NOT = { war_with = SPA }
		NOT = { the_low_countries = { owned_by = SPA } }
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 2
			mil = 4
		}
		modifier = {
			factor = 2
			mil = 5
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			remove_claim = ROOT
		}
	}
	effect = {
		add_war_exhaustion = -5
		add_army_tradition = 25
	}
}
