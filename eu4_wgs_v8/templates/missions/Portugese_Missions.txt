anglo_portuguese_relations = {

	type = country
	
	category = DIP
	
	allow = {
		tag = POR
		exists = ENG
		government = monarchy
		NOT = { has_opinion = { who = ENG value = 50 } }
		ENG = {
			is_subject = no
			government = monarchy
		}
		NOT = { marriage_with = ENG }
		NOT = { war_with = ENG }
	}
	abort = {
		OR = {
			NOT = { government = monarchy }
			war_with = ENG
			NOT = { exists = ENG }
			ENG = {
				is_subject = yes
				NOT = { government = monarchy }
			}
		}
	}
	success = {
		ENG = { has_opinion = { who = POR value = 100 } }
		marriage_with = ENG
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = ENG value = 0 } }
		}
	}
	effect = {
		add_prestige = 10
	}
}
