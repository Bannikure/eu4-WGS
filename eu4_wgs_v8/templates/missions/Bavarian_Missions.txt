bavaria_austria_relations = {
	
	type = country

	category = DIP
	
	allow = {
		tag = BAV
		exists = HAB
		is_subject = no
		government = monarchy
		NOT = { war_with = HAB }
		NOT = { marriage_with = HAB }
		HAB = {
			is_subject = no
			government = monarchy
		}
	}
	abort = {
		OR = {
			is_subject = yes
			NOT = { government = monarchy }
			war_with = HAB
			HAB = {
				OR = {
					is_subject = yes
					NOT = { government = monarchy }
				}	
			}
		}
	}
	success = {
		marriage_with = HAB
		HAB = { has_opinion = { who = BAV value = 100 } }
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = HAB value = 0 } }
		}
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = HAB value = -100 } }
		}
	}
	effect = {
		add_stability = 1
	}
}
